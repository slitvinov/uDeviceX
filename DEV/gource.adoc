= gource command

----
gource -s 0.1 --file-filter 'test_data/' \
	      --file-filter 'poc/' \
	      --file-filter 'tcdf/' \
	      --file-filter 'cmd/'  \
	      --date-format '%d %b %Y'
----

`-o g.ppm` to dump

----
ffmpeg -y -r 60 -f image2pipe -vcodec ppm -i g.ppm -vcodec libx264 -preset ultrafast -pix_fmt yuv420p -crf 1 -threads 0 -bf 0 g.mp4
----
